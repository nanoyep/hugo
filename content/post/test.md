$$
\begin{aligned}
c^d &\equiv (m^e)^d \pmod{n} \\
&\equiv m^{ed} \pmod{n} \\
&\equiv m^{1 + k\varphi(n)} \pmod{n} \quad (\text{因为 } ed \equiv 1 \pmod{\varphi(n)}) \\
&\equiv m \cdot (m^{\varphi(n)})^k \pmod{n} \\
&\equiv m \cdot 1^k \pmod{n} \quad (\text{由欧拉定理 } m^{\varphi(n)} \equiv 1 \pmod{n}) \\
&\equiv m \pmod{n}
\end{aligned}
$$
